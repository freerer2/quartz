DROP TABLE IF EXISTS ATTENDANCE.ACT_LIST CASCADE;  
  
CREATE TABLE ATTENDANCE.ACT_LIST  
(

COMMENT ON COLUMN ATTENDANCE.ACT_LIST.CRT_DTTM IS '생성일시';  
  
COMMENT ON TABLE ATTENDANCE.ACT_LIST IS '행동목록';
